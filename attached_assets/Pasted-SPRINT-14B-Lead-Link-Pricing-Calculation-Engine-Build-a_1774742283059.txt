interface LeadLinkPricingConfig {
  hourlyRate: number;
  minimumTicket: number;
  taxRate: number;
  serviceTypes: ServiceType[];
  goodOptionId: string;
  betterOptionId: string;
  bestOptionId: string;
  frequencyDiscounts: FrequencyDiscounts;
  addOnPrices: AddOnPrices;
}

interface QuoteInputs {
  bedrooms: number;           // 0-10
  bathrooms: number;          // 1-6
  sqft: number | null;        // optional, null if not provided
  cleaningType: "standard" | "deep" | "moveinout" | "recurring";
  frequency: "onetime" | "weekly" | "biweekly" | "monthly";
  conditionLevel: "excellent" | "good" | "average" | "dirty";
  petCount: number;           // 0-5+
  addOns: (keyof AddOnPrices)[];
  usingDefaultPricing: boolean;
}

interface PriceEstimate {
  lowEstimate: number;
  highEstimate: number;
  midEstimate: number;        // (low + high) / 2, rounded to $5
  serviceTypeName: string;    // "Standard Clean", "Deep Clean" etc
  frequencyLabel: string;     // "One-time", "Every 2 weeks" etc
  frequencyDiscountPercent: number;
  addOnsSubtotal: number;
  breakdown: PriceBreakdown;
  confidence: "high" | "medium" | "low";
}

interface PriceBreakdown {
  baseLabor: number;
  conditionAdjustment: number;
  petFee: number;
  addOnsTotal: number;
  frequencyDiscount: number;
  subtotal: number;
  tax: number;
  total: number;
}

CALCULATION FUNCTION:

export function calculateLeadLinkEstimate(
  inputs: QuoteInputs,
  config: LeadLinkPricingConfig
): PriceEstimate {

  // STEP 1: DETERMINE SERVICE TYPE
  // Map cleaningType to the business owner's 
  // configured service type multiplier:
  
  const serviceTypeMap = {
    standard: config.goodOptionId,
    deep: config.betterOptionId,
    moveinout: config.bestOptionId,
    recurring: config.goodOptionId,
  };
  
  const serviceType = config.serviceTypes.find(
    st => st.id === serviceTypeMap[inputs.cleaningType]
  ) || config.serviceTypes[0];
  
  const multiplier = serviceType?.multiplier || 1.0;

  // STEP 2: CALCULATE BASE LABOR HOURS
  // Use sqft if provided, fall back to 
  // bedroom/bathroom estimation:
  
  let estimatedHours: number;
  
  if (inputs.sqft && inputs.sqft > 0) {
    // sqft-based calculation:
    // Base: 1 hour per 400 sqft for standard
    estimatedHours = inputs.sqft / 400;
  } else {
    // bedroom/bathroom-based calculation:
    // Base hours lookup table:
    const bedroomHours: Record<number, number> = {
      0: 0.75,  // studio
      1: 1.0,
      2: 1.5,
      3: 2.0,
      4: 2.5,
      5: 3.0,
      6: 3.5,
    };
    const bathroomHours: Record<number, number> = {
      1: 0.5,
      1.5: 0.65,
      2: 0.75,
      2.5: 0.9,
      3: 1.0,
      3.5: 1.15,
      4: 1.25,
    };
    
    const bedHours = bedroomHours[
      Math.min(inputs.bedrooms, 6)
    ] || 2.0;
    
    const bathHours = bathroomHours[
      Math.min(inputs.bathrooms, 4)
    ] || 0.75;
    
    estimatedHours = bedHours + bathHours;
  }
  
  // STEP 3: APPLY SERVICE TYPE MULTIPLIER
  estimatedHours = estimatedHours * multiplier;

  // STEP 4: BASE LABOR COST
  const baseLabor = estimatedHours * config.hourlyRate;

  // STEP 5: CONDITION ADJUSTMENT
  const conditionMultipliers = {
    excellent: 0.90,  // 10% discount — easy job
    good:      1.00,  // standard
    average:   1.15,  // 15% more
    dirty:     1.35,  // 35% more
  };
  
  const conditionMultiplier = 
    conditionMultipliers[inputs.conditionLevel] || 1.0;
  
  const conditionAdjustment = 
    baseLabor * conditionMultiplier - baseLabor;

  // STEP 6: PET FEE
  // $15 per pet, max $45 (3 pets)
  const petFee = Math.min(inputs.petCount * 15, 45);

  // STEP 7: ADD-ONS
  const addOnsTotal = inputs.addOns.reduce((sum, addOn) => {
    return sum + (config.addOnPrices[addOn] || 0);
  }, 0);

  // STEP 8: SUBTOTAL BEFORE DISCOUNT
  const subtotalBeforeDiscount = 
    baseLabor + 
    conditionAdjustment + 
    petFee + 
    addOnsTotal;

  // STEP 9: FREQUENCY DISCOUNT
  const frequencyDiscountMap = {
    onetime:  0,
    weekly:   config.frequencyDiscounts.weekly / 100,
    biweekly: config.frequencyDiscounts.biweekly / 100,
    monthly:  config.frequencyDiscounts.monthly / 100,
  };
  
  const discountPercent = 
    frequencyDiscountMap[inputs.frequency] || 0;
  
  const frequencyDiscount = 
    subtotalBeforeDiscount * discountPercent;

  // STEP 10: SUBTOTAL AFTER DISCOUNT
  const subtotal = subtotalBeforeDiscount - frequencyDiscount;

  // STEP 11: APPLY MINIMUM TICKET
  const subtotalWithMinimum = Math.max(
    subtotal, 
    config.minimumTicket
  );

  // STEP 12: TAX
  const tax = subtotalWithMinimum * (config.taxRate / 100);

  // STEP 13: TOTAL
  const total = subtotalWithMinimum + tax;

  // STEP 14: CALCULATE RANGE
  // Range width depends on confidence:
  // - If sqft provided: tighter range (more accurate)
  // - If using default pricing: wider range
  // - If condition is dirty: wider range (unpredictable)
  
  let rangeWidth: { low: number; high: number };
  
  if (inputs.usingDefaultPricing) {
    rangeWidth = { low: 0.70, high: 1.50 };
  } else if (inputs.sqft && inputs.sqft > 0) {
    rangeWidth = { low: 0.90, high: 1.15 };
  } else if (inputs.conditionLevel === "dirty") {
    rangeWidth = { low: 0.82, high: 1.28 };
  } else {
    rangeWidth = { low: 0.85, high: 1.25 };
  }
  
  // Round to nearest $5
  const roundTo5 = (n: number) => Math.round(n / 5) * 5;
  
  const lowEstimate = roundTo5(total * rangeWidth.low);
  const highEstimate = roundTo5(total * rangeWidth.high);
  const midEstimate = roundTo5((lowEstimate + highEstimate) / 2);

  // STEP 15: CONFIDENCE LEVEL
  let confidence: "high" | "medium" | "low";
  if (inputs.usingDefaultPricing) {
    confidence = "low";
  } else if (inputs.sqft && inputs.sqft > 0) {
    confidence = "high";
  } else {
    confidence = "medium";
  }

  // STEP 16: FREQUENCY LABEL
  const frequencyLabels = {
    onetime:  "One-time clean",
    weekly:   "Weekly recurring",
    biweekly: "Every 2 weeks",
    monthly:  "Monthly recurring",
  };

  return {
    lowEstimate,
    highEstimate,
    midEstimate,
    serviceTypeName: serviceType?.name || "Standard Clean",
    frequencyLabel: frequencyLabels[inputs.frequency],
    frequencyDiscountPercent: discountPercent * 100,
    addOnsSubtotal: addOnsTotal,
    breakdown: {
      baseLabor: roundTo5(baseLabor),
      conditionAdjustment: roundTo5(conditionAdjustment),
      petFee,
      addOnsTotal,
      frequencyDiscount: roundTo5(frequencyDiscount),
      subtotal: roundTo5(subtotalWithMinimum),
      tax: roundTo5(tax),
      total: roundTo5(total),
    },
    confidence,
  };
}

REAL-TIME UPDATES ON MICROSITE
═══════════════════════════════

The price range must update in real time as 
the customer fills in Step 2 fields.

IMPLEMENTATION:
- Call calculateLeadLinkEstimate() on every 
  field change in Step 2
- Debounce: 150ms (fast enough to feel live)
- Animate number changes:
  When lowEstimate or highEstimate changes, 
  briefly flash the number:
  transition: color 200ms ease
  Momentarily set color to primary blue, 
  then fade back to white
  
- Update the persistent price bar at top:
  "$[LOW] – $[HIGH]" updates smoothly
  
- If range narrows (more info = more accurate):
  Show a brief "Getting more accurate..." 
  tooltip near the price bar
  Show for 1.5 seconds then fade out

- Never show a price below the minimumTicket:
  If calculated lowEstimate < minimumTicket:
    lowEstimate = minimumTicket
    highEstimate = Math.max(
      highEstimate, 
      minimumTicket * 1.3
    )

═══════════════════════════════
VALIDATION & EDGE CASES
═══════════════════════════════

Handle ALL of these edge cases explicitly:

1. Studio apartment (0 bedrooms):
   Use bedroomHours[0] = 0.75
   Do not error or return $0

2. 5+ bedrooms:
   Cap at 6 bedrooms for calculation
   Add a note: "For larger homes, exact pricing 
   will be confirmed by [BusinessName]"

3. Hourly rate is 0 or missing:
   Fall back to $40/hour default
   Flag as usingDefaultPricing = true
   Log error server-side

4. minimumTicket is 0 or missing:
   Fall back to $100 minimum
   Flag as usingDefaultPricing = true

5. All add-on prices are 0:
   Treat add-ons as not configured
   Do not add $0 line items to the estimate

6. taxRate is 0:
   Valid — many cleaning businesses don't 
   charge tax. Do not flag as misconfigured.

7. frequencyDiscounts all 0:
   Valid — business may not offer discounts.
   Do not flag as misconfigured.

8. serviceTypes array is empty:
   Fall back to multiplier of 1.0 for all types
   Flag as usingDefaultPricing = true

9. cleaningType = "moveinout" but no bestOptionId:
   Fall back to betterOptionId, then goodOptionId

10. Very large sqft (>5000):
    Cap calculation at 5000 sqft equivalent hours
    Add note: "For commercial or very large 
    properties, [BusinessName] will provide 
    a custom quote"
